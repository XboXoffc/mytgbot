## /game
  just a telegram games
  - flags:
    - -b or -basket — basketball
    - -d or -darts — darts
    - -f or -football — soccer
    - -bl or -bowling — bowling
    - -s or -slots — arcade slots

  - flags:
    - -add or -a -- additional info
    - -p -- ???
